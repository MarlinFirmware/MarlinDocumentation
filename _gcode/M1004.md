    tag: H

    values:
      -
        tag: temp
        type: float
  -
    tag: S
    optional: true
    description: Save the mesh to EEPROM in the given slot.
    values:
      -
        tag: slot
        type: int
        
examples:

    code: M1004 H150 B60 S0
